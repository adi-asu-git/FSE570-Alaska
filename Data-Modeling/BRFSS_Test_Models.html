10 - Model 1: Weighted Logistic Regression with Rubin Rules¶

Categorical encoding throughout. Three interactions: Age x Race, Income x Education, Sex x Age. Rubin rules pool m=5 fits via: Q_bar = mean(coefs), T = U_bar + (1+1/m)*B.

In [21]:
def encode_logit(df_imp):
    return encoder.transform(df_imp[CAT_FEATURES]).astype(np.float32)

def rubin_pool(coefs_list, vars_list):
    m     = len(coefs_list)
    Q_bar = np.mean(coefs_list, axis=0)
    U_bar = np.mean(vars_list,  axis=0)
    B     = np.var(coefs_list,  axis=0, ddof=1)
    T     = U_bar + (1 + 1/m) * B
    return Q_bar, T

class Obesitynn(nn.Module):
    def __init__(self, input_dim, n_states=NUM_STATES, state_emb_dim=STATE_EMB_DIM,
                 hidden=None, dropout=DROPOUT):
        super().__init__()
        if hidden is None: hidden = HIDDEN_DIMS
        self.state_emb = nn.Embedding(n_states, state_emb_dim)
        full_dim = input_dim + state_emb_dim
        self.block1 = nn.Sequential(
            nn.Linear(full_dim, hidden[0]), nn.BatchNorm1d(hidden[0]),
            nn.ReLU(), nn.Dropout(dropout))
        self.block2 = nn.Sequential(
            nn.Linear(hidden[0], hidden[1]), nn.BatchNorm1d(hidden[1]),
            nn.ReLU(), nn.Dropout(dropout))
        self.block3 = nn.Sequential(
            nn.Linear(hidden[1], hidden[2]), nn.BatchNorm1d(hidden[2]),
            nn.ReLU(), nn.Dropout(dropout))
        self.skip   = nn.Linear(full_dim, hidden[2], bias=False)
        self.output = nn.Linear(hidden[2], 1)
        for m in self.modules():
            if isinstance(m, nn.Linear):
                nn.init.kaiming_normal_(m.weight, nonlinearity='relu')
                if m.bias is not None: nn.init.zeros_(m.bias)

    def forward(self, x_demo, state_idx):
        s = self.state_emb(state_idx)
        x = torch.cat([x_demo, s], dim=1)
        h = self.block3(self.block2(self.block1(x)))
        return self.output(h + self.skip(x)).squeeze(-1)

    def get_components(self, x_demo, state_idx):
        s   = self.state_emb(state_idx)
        x   = torch.cat([x_demo, s], dim=1)
        lin = self.skip(x)
        nln = self.block3(self.block2(self.block1(x)))
        tot = self.output(lin + nln).squeeze(-1)
        return lin, nln, tot

def weighted_bce(logits, targets, weights):
    loss = F.binary_cross_entropy_with_logits(logits, targets, reduction='none')
    return (loss * weights).mean()

def make_loader(X, df_split, shuffle=True):
    y = torch.tensor(df_split['obese'].values, dtype=torch.float32)
    w = torch.tensor(df_split['_LLCPWT_s'].values, dtype=torch.float32)
    s = torch.tensor(df_split['state_idx'].values, dtype=torch.long)  # pre-encoded
    ds = TensorDataset(torch.tensor(X, dtype=torch.float32), y, w, s)
    return DataLoader(ds, batch_size=BATCH_SIZE, shuffle=shuffle,
                      num_workers=0, pin_memory=(DEVICE.type=='cuda'))

def train_epoch(model, loader, optimizer):
    model.train()
    total_loss, total_n = 0.0, 0
    for X_b, y_b, w_b, s_b in loader:
        X_b,y_b,w_b,s_b = X_b.to(DEVICE),y_b.to(DEVICE),w_b.to(DEVICE),s_b.to(DEVICE)
        optimizer.zero_grad()
        logits = model(X_b, s_b)
        loss   = weighted_bce(logits, y_b, w_b)
        loss.backward()
        nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
        optimizer.step()
        total_loss += loss.item() * len(y_b)
        total_n    += len(y_b)
    return total_loss / total_n

@torch.no_grad()
def evaluate_loader(model, loader):
    model.eval()

class TemperatureScaler(nn.Module):
    def __init__(self, model):
        super().__init__()
        self.model = model
        self.T     = nn.Parameter(torch.ones(1, device=DEVICE))

    @torch.no_grad()
    def fit(self, val_loader, max_iter=50):
        self.model.eval()
        all_logits, all_y, all_w = [], [], []
        for X_b, y_b, w_b, s_b in val_loader:
            logits = self.model(X_b.to(DEVICE), s_b.to(DEVICE))
            all_logits.append(logits)
            all_y.append(y_b.to(DEVICE))
            all_w.append(w_b.to(DEVICE))
        logits = torch.cat(all_logits)
        y      = torch.cat(all_y)
        w      = torch.cat(all_w)

        opt = torch.optim.LBFGS([self.T], max_iter=max_iter, lr=0.01)
        def closure():
            opt.zero_grad()
            loss = weighted_bce(logits / self.T, y, w)
            loss.backward()
            return loss
        opt.step(closure)
        return self

    def predict_proba(self, X, s):
        self.model.eval()
        with torch.no_grad():
            logits = self.model(X.to(DEVICE), s.to(DEVICE))
            return torch.sigmoid(logits / self.T).cpu().numpy()

def compute_shift_weights(df_imp):
    rng = np.random.RandomState(SEED)
    n   = len(df_imp)

    for col, dist in ACS_MARGINALS.items():
        codes = list(dist.keys())

        probs /= probs.sum()
        acs_sim[col] = rng.choice(codes, size=n, p=probs)
